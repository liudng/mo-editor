46bool Document::load(const QString &path)
47{
48 QFile file(path);
49 if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
50 return false;
51 }
52 QTextStream in(&file);
53 in.setEncoding(QStringConverter::Utf8);
54 content_ = in.readAll();
55 file.close();
56
57 setFilePath(path);
58 setModified(false);
