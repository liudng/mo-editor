97CodeEditor *FindReplaceDock::currentEditor() const
98{
99 // The dock's parent is the MainWindow; find its QTabWidget.
100 auto *mw = parentWidget();
101 if (!mw) {
102 return nullptr;
103 }
104 auto *tw = mw->findChild<QTabWidget *>();
105 if (!tw) {
106 return nullptr;
107 }
108 return qobject_cast<CodeEditor *>(tw->currentWidget());

143void FindReplaceDock::replace(CodeEditor *editor)
144{
145 if (!editor || searchText().isEmpty()) {
146 return;
147 }
148 auto cursor = editor->textCursor();
149 if (cursor.hasSelection() && cursor.selectedText() == searchText()) {
150 cursor.insertText(replaceText());
151 }
152 // Advance to next match.
153 editor->findNext(searchText());
154 updateStatus(tr("Replaced one occurrence"));
155}
156
157void FindReplaceDock::replaceAll(CodeEditor *editor)
158{
159 if (!editor || searchText().isEmpty()) {
160 return;
161 }
162 const QString findText = searchText();
163 const QString replText = replaceText();
164 int count = 0;
165
166 QTextCursor cursor(editor->document());
167 cursor.beginEditBlock();
168 while (!cursor.isNull()) {
169 cursor = editor->document()->find(findText, cursor);
170 if (cursor.isNull()) {
171 break;
172 }
173 cursor.insertText(replText);
174 ++count;
175 }
176 cursor.endEditBlock();
177
178 updateStatus(tr("Replaced %n occurrence(s)", "", count));
179}
