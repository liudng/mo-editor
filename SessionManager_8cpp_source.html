25void SessionManager::saveSession(const SessionState &state)

30 settings.beginGroup(QStringLiteral("session"));

39std::optional<SessionState> SessionManager::loadSession() const
40{
41 QSettings settings(settingsPath(), QSettings::IniFormat);
42
43 settings.beginGroup(QStringLiteral("session"));
44 if (!settings.contains(QStringLiteral("openFiles"))) {
45 settings.endGroup();
46 return std::nullopt;
47 }
48 SessionState state;
49 state.openFiles = settings.value(QStringLiteral("openFiles")).toStringList();
50 state.cursorPositions = settings.value(QStringLiteral("cursorPositions")).toStringList();
51 state.windowGeometry = settings.value(QStringLiteral("windowGeometry")).toByteArray();
