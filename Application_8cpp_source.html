63void Application::loadTranslation()
64{
65 // Qt's own translation (dialogs, standard shortcuts).
66 static QTranslator qtTranslator;
67 if (qtTranslator.load(QLocale::system(), QStringLiteral("qt"), QStringLiteral("_"),
68 QLibraryInfo::path(QLibraryInfo::TranslationsPath))) {
69 installTranslator(&qtTranslator);
70 }
71
72 // Application translation; load zh_CN resource when locale is Chinese.
73 static QTranslator appTranslator;
74 const auto locale = QLocale().name();
75 if (locale.startsWith(QStringLiteral("zh"))) {
76 if (appTranslator.load(QStringLiteral(":/i18n/mo_zh_CN.qm"))) {
77 installTranslator(&appTranslator);
78 } else {
79 mo::core::Logger::warning("Failed to load zh_CN translation resource");
80 }
81 }
82}
