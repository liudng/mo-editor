21bool SingleInstance::tryLock()
22{
23 // Primary mechanism: register a well-known D-Bus service on the session bus.
24 // On Wayland/Linux this both detects a previous instance and provides an
25 // activation path for future instances.
26 auto bus = QDBusConnection::sessionBus();
27 if (bus.isConnected()) {
28 if (bus.registerService(mo::core::constants::kDBusServiceName)) {
29 locked_ = true;
30 mo::core::Logger::info("Single instance acquired via D-Bus");
31 return true;
32 }
33 mo::core::Logger::warning(
34 "D-Bus service already registered; another instance is running");
35 return false;
36 }
37
38 // Fallback mechanism: a shared memory segment.
39 if (!sharedMemory_.create(1)) {
40 if (sharedMemory_.attach()) {
41 sharedMemory_.detach();
42 }
43 mo::core::Logger::warning(
44 "Another instance is already running (QSharedMemory fallback)");
45 return false;
46 }
47
48 locked_ = true;
49 return true;
50}
