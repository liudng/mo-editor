24SettingsDialog::SettingsDialog(QWidget *parent) : QDialog(parent)
25{
26 setWindowTitle(tr("Preferences"));
27
28 fontCombo_ = new QFontComboBox(this);
29 fontSizeSpin_ = new QSpinBox(this);
30 fontSizeSpin_->setRange(6, 48);
31
32 themeCombo_ = new QComboBox(this);
33 themeCombo_->addItem(tr("System"), QStringLiteral("System"));
34 themeCombo_->addItem(tr("Light"), QStringLiteral("Light"));
35 themeCombo_->addItem(tr("Dark"), QStringLiteral("Dark"));
36
37 iconThemeCombo_ = new QComboBox(this);
38 iconThemeCombo_->setToolTip(tr("Fallback icon theme used when the system "
39 "theme is missing icons (e.g. Adwaita)."));
40 // Populate with icon themes found in standard search paths.
41 iconThemeCombo_->addItem(tr("System Default"), QString());
42 const auto iconPaths = QIcon::themeSearchPaths();
43 QSet<QString> seen;
44 for (const auto &path : iconPaths) {
45 QDir dir(path);
46 if (!dir.exists()) {
47 continue;
48 }
49 QDirIterator it(path, QDir::Dirs | QDir::NoDotAndDotDot);
50 while (it.hasNext()) {
51 it.next();
52 // Only list directories that look like icon themes (contain index.theme).
53 if (QFileInfo::exists(it.filePath() + QStringLiteral("/index.theme"))) {
54 const auto name = it.fileName();
55 if (!seen.contains(name)) {
56 seen.insert(name);
57 iconThemeCombo_->addItem(name, name);
58 }
59 }
60 }
61 }
62
63 tabWidthSpin_ = new QSpinBox(this);
64 tabWidthSpin_->setRange(1, 16);
65
66 lineNumbersCheck_ = new QCheckBox(tr("Show line numbers"), this);
67 autoIndentCheck_ = new QCheckBox(tr("Auto indent"), this);
68
69 auto *form = new QFormLayout;
70 form->addRow(tr("Font:"), fontCombo_);
71 form->addRow(tr("Font size:"), fontSizeSpin_);

90void SettingsDialog::load()
91{
92 auto &s = mo::core::Settings::instance();
93 fontCombo_->setCurrentFont(QFont(s.fontFamily()));
94 fontSizeSpin_->setValue(s.fontSize());
95
96 const auto theme = s.theme();
97 for (int i = 0; i < themeCombo_->count(); ++i) {
98 if (themeCombo_->itemData(i).toString() == theme) {
99 themeCombo_->setCurrentIndex(i);
100 break;
101 }
102 }
103
104 const auto iconTheme = s.iconTheme();
105 bool found = false;
106 for (int i = 0; i < iconThemeCombo_->count(); ++i) {
107 const auto data = iconThemeCombo_->itemData(i).toString();
108 if (data == iconTheme || (data.isEmpty() && iconTheme.isEmpty())) {
109 iconThemeCombo_->setCurrentIndex(i);
110 found = true;
111 break;
112 }
113 }
114 if (!found) {
115 // The configured icon theme is not installed; show it as a custom entry.
116 iconThemeCombo_->insertItem(0, iconTheme, iconTheme);
117 iconThemeCombo_->setCurrentIndex(0);
118 }
119
120 tabWidthSpin_->setValue(s.tabWidth());
121 lineNumbersCheck_->setChecked(s.showLineNumbers());
122 autoIndentCheck_->setChecked(s.autoIndent());
123}

125void SettingsDialog::save()
126{
127 auto &s = mo::core::Settings::instance();
128 s.setFontFamily(fontCombo_->currentFont().family());
129 s.setFontSize(fontSizeSpin_->value());
130 s.setTheme(themeCombo_->currentData().toString());
131 s.setIconTheme(iconThemeCombo_->currentData().toString());
132 s.setTabWidth(tabWidthSpin_->value());
133 s.setShowLineNumbers(lineNumbersCheck_->isChecked());
134 s.setAutoIndent(autoIndentCheck_->isChecked());
135 s.save();
136}
