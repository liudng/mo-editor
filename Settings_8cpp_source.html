19Settings::Settings()
20{
21 const auto configDir = QStandardPaths::writableLocation(QStandardPaths::AppConfigLocation);
22 QDir().mkpath(configDir);
23 const auto path = configDir + QDir::separator() + mo::core::constants::kSettingsFile;
24 settings_ = std::make_unique<QSettings>(path, QSettings::IniFormat);
25 // Qt6 reads/writes INI files as UTF-8 by default; no codec setup needed.
