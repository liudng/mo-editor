95int CodeEditor::lineNumberAreaWidth()
96{
97 if (!showLineNumbers_) {
98 return 0;
99 }
100 int digits = 1;
101 int max = qMax(1, blockCount());
102 while (max >= 10) {
103 max /= 10;
104 ++digits;
105 }
106 const int space = 3 + fontMetrics().horizontalAdvance(QLatin1Char('9')) * digits;
107 return space;
108}

136void CodeEditor::keyPressEvent(QKeyEvent *event)
137{
138 // Auto-indent: when pressing Enter, copy leading whitespace from current line.
139 if (mo::core::Settings::instance().autoIndent()
140 && event->key() == Qt::Key_Return && !(event->modifiers() & Qt::ShiftModifier)) {
141 auto cursor = textCursor();
142 const QString lineText = cursor.block().text();
143 QString indent;
144 for (const auto ch : lineText) {
145 if (ch == QLatin1Char(' ') || ch == QLatin1Char('\t')) {
146 indent += ch;
147 } else {
148 break;
149 }
150 }
151 QPlainTextEdit::keyPressEvent(event);
152 if (!indent.isEmpty()) {
153 cursor = textCursor();
154 cursor.insertText(indent);
155 }
156 return;
157 }
158
159 // Tab key: insert spaces according to tabWidth setting.
160 if (event->key() == Qt::Key_Tab && !(event->modifiers() & Qt::ControlModifier)) {
161 const int tw = mo::core::Settings::instance().tabWidth();
162 insertPlainText(QString(tw, QLatin1Char(' ')));
163 return;
164 }
165
166 QPlainTextEdit::keyPressEvent(event);
167}

169void CodeEditor::highlightCurrentLine()
170{
171 QList<QTextEdit::ExtraSelection> extraSelections;
172 if (!isReadOnly()) {
173 QTextEdit::ExtraSelection selection;
174 QColor lineColor;
175 if (highlightingTheme_.isValid()) {
176 lineColor = QColor::fromRgba(highlightingTheme_.editorColor(
177 KSyntaxHighlighting::Theme::EditorColorRole::CurrentLine));
178 }
179 if (!lineColor.isValid()) {
180 lineColor = palette().color(QPalette::AlternateBase);
181 }
182 selection.format.setBackground(lineColor);
183 selection.format.setProperty(QTextFormat::FullWidthSelection, true);
184 selection.cursor = textCursor();
185 selection.cursor.clearSelection();
186 extraSelections.append(selection);
187 }
188 setExtraSelections(extraSelections);
189}

191void CodeEditor::lineNumberAreaPaintEvent(QPaintEvent *event)
192{
193 QPainter painter(lineNumberArea_);
194
195 QColor bgColor;
196 QColor fgColor;
197 if (highlightingTheme_.isValid()) {
198 bgColor = QColor::fromRgba(highlightingTheme_.editorColor(
199 KSyntaxHighlighting::Theme::EditorColorRole::IconBorder));
200 fgColor = QColor::fromRgba(highlightingTheme_.editorColor(
201 KSyntaxHighlighting::Theme::EditorColorRole::LineNumbers));
202 }
203 if (!bgColor.isValid()) {
204 bgColor = palette().color(QPalette::Window);
205 }
206 if (!fgColor.isValid()) {
207 fgColor = palette().color(QPalette::BrightText);
208 if (!fgColor.isValid()) {
209 fgColor = palette().color(QPalette::Text);
210 }
211 }
212
213 painter.fillRect(event->rect(), bgColor);
214
215 QTextBlock block = firstVisibleBlock();
216 int blockNumber = block.blockNumber();
217 int top = qRound(blockBoundingGeometry(block).translated(contentOffset()).top());
218 int bottom = top + qRound(blockBoundingRect(block).height());
219
220 painter.setPen(fgColor);
221 while (block.isValid() && top <= event->rect().bottom()) {
222 if (block.isVisible() && bottom >= event->rect().top()) {
223 const QString number = QString::number(blockNumber + 1);
224 painter.drawText(0, top, lineNumberArea_->width() - 2,
225 fontMetrics().height(), Qt::AlignRight, number);
226 }
227 block = block.next();
228 top = bottom;
229 bottom = top + qRound(blockBoundingRect(block).height());
230 ++blockNumber;
231 }
232}

234bool CodeEditor::loadFile(const QString &path)
235{
236 QFile file(path);
237 if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
238 mo::core::Logger::warning("Failed to open: " + path);
239 return false;
240 }
241 QTextStream in(&file);
242 in.setEncoding(QStringConverter::Utf8);
243 setPlainText(in.readAll());
244 file.close();
245
246 filePath_ = path;
247 setLanguage(QFileInfo(path).suffix());
248 document()->setModified(false);
249 updateTabTitle();
250 return true;
251}

261bool CodeEditor::saveAs(const QString &path)
262{
263 QSaveFile file(path);
264 if (!file.open(QIODevice::WriteOnly | QIODevice::Text)) {
265 return false;
266 }
267 QTextStream out(&file);
268 out.setEncoding(QStringConverter::Utf8);
269 out << toPlainText();
270 out.flush();
271 if (file.commit()) {
272 filePath_ = path;
273 document()->setModified(false);
274 updateTabTitle();
275 return true;
276 }
277 return false;
278}
